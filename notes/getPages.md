export function getPages(currentPage, maxPage, limit) {
    // 1. –ù–æ—Ä–º–∞–ª–∏–∑—É–µ–º –≤—Ö–æ–¥–Ω—ã–µ –¥–∞–Ω–Ω—ã–µ
    currentPage = Math.max(1, Math.min(maxPage, currentPage));  // currentPage –æ—Ç 1 –¥–æ maxPage
    limit = Math.min(maxPage, limit);  // limit –Ω–µ –±–æ–ª—å—à–µ maxPage

    // 2. –í—ã—á–∏—Å–ª—è–µ–º –¥–∏–∞–ø–∞–∑–æ–Ω —Å—Ç—Ä–∞–Ω–∏—Ü –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è
    let start = Math.max(1, currentPage - Math.floor(limit / 2));
    let end = start + limit - 1;

    // 3. –ö–æ—Ä—Ä–µ–∫—Ç–∏—Ä—É–µ–º –≥—Ä–∞–Ω–∏—Ü—ã, –µ—Å–ª–∏ –≤—ã—Ö–æ–¥–∏–º –∑–∞ –ø—Ä–µ–¥–µ–ª—ã
    if (end > maxPage) {
        end = maxPage;
        start = Math.max(1, end - limit + 1);
    }

    // 4. –§–æ—Ä–º–∏—Ä—É–µ–º –º–∞—Å—Å–∏–≤ —Å—Ç—Ä–∞–Ω–∏—Ü
    const pages = [];
    for (let i = start; i <= end; i++) {
        pages.push(i);
    }

    return pages;
}
